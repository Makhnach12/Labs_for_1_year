#include <iostream>
#include<math.h>
using namespace std;
int main()
	{
	int n; int n1; double a; double s;
	cout << "Write a number of members of the sequence\n" << "n=";
	cin >> n;//n-количество чисел в последовательности
	while (std::cin.fail())
	{
		std::cin.clear();
		std::cin.ignore(32767, '\n');
		cout << "Please, write correct number\n";
		cin >> n;
	}

	s = 0;
	n1 = 0;
	while (n > 0)
	{
		n1++;//—четчик дл€ пор€дковых номеров
		cout << "Write the a number " << n1 << " of sequense:";
		cin >> a;//вбиваем числа последовательности
		while (std::cin.fail())
		{
			std::cin.clear();
			std::cin.ignore(32767, '\n');
			cout << "Please, write correct number\n";
			cin >> a;
		}

		if (abs(a) < n1 * n1)//провер€ем выполнение услови€
		s= s+a;
		n=n-1;
	}
	cout<<"The sum is "<<s;
	return 0;
	}
